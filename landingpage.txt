        // --- Image Slideshow Logic ---
        const images = [
            'https://images.unsplash.com/photo-1473496169904-658ba7c44d8a?q=80&w=2070&auto=format&fit=crop',
            'https://images.unsplash.com/photo-1598554747476-38428a3c99b5?q=80&w=1887&auto=format&fit=crop',
            'https://images.unsplash.com/photo-1555982339-653e8d44c1ab?q=80&w=1887&auto=format&fit=crop',
            'https://images.unsplash.com/photo-1620799140408-edc6d5f9650d?q=80&w=1972&auto=format&fit=crop',
            'https://images.unsplash.com/photo-1588702547919-26089e690ecc?q=80&w=2070&auto=format&fit=crop',
            'https://images.unsplash.com/photo-1523170335258-f5ed11844a49?q=80&w=2080&auto=format&fit=crop',
            'https://images.unsplash.com/photo-1512496015851-a90138ba7a6a?q=80&w=1887&auto=format&fit=crop',
            'https://images.unsplash.com/photo-1616738343113-d4543d0473df?q=80&w=1887&auto=format&fit=crop',
            'https://images.unsplash.com/photo-1594223274512-ad4803739b7c?q=80&w=1887&auto=format&fit=crop',
            'https://images.unsplash.com/photo-1543163521-1bf539c55dd2?q=80&w=1760&auto=format&fit=crop'
        ];

        images.forEach(src => { new Image().src = src; });

        const bgLayer1 = document.getElementById('bg-layer-1');
        const bgLayer2 = document.getElementById('bg-layer-2');
        let currentImageIndex = 0;

        bgLayer1.style.backgroundImage = `url(${images[currentImageIndex]})`;

        setInterval(() => {
            currentImageIndex = (currentImageIndex + 1) % images.length;
            const nextImage = images[currentImageIndex];
            const activeLayer = bgLayer1.style.opacity == '0' ? bgLayer1 : bgLayer2;
            const hiddenLayer = bgLayer1.style.opacity == '0' ? bgLayer2 : bgLayer1;

            activeLayer.style.backgroundImage = `url(${nextImage})`;
            activeLayer.style.opacity = '1';
            hiddenLayer.style.opacity = '0';
        }, 3000);

        // --- Sticky Scroll Animation Logic ---
        window.addEventListener('scroll', () => {
            const heroContent = document.querySelector('#hero-section .content-wrapper');
            const whyUsContent = document.querySelector('#why-us-section .content-wrapper');
            
            const scrollY = window.scrollY;
            const vh = window.innerHeight;

            const heroProgress = Math.min(1, scrollY / vh);
            heroContent.style.opacity = 1 - heroProgress * 1.5;
            heroContent.style.filter = `blur(${heroProgress * 20}px)`;
            heroContent.style.transform = `scale(${1 - heroProgress * 0.1})`;

            const whyUsScrollStart = vh;
            if (scrollY > whyUsScrollStart) {
                const whyUsProgress = Math.min(1, (scrollY - whyUsScrollStart) / vh);
                whyUsContent.style.opacity = 1 - whyUsProgress * 1.5;
                whyUsContent.style.filter = `blur(${whyUsProgress * 20}px)`;
                whyUsContent.style.transform = `scale(${1 - whyUsProgress * 0.1})`;
            } else {
                whyUsContent.style.opacity = 1;
                whyUsContent.style.filter = 'blur(0px)';
                whyUsContent.style.transform = 'scale(1)';
            }
        });
    </script>
